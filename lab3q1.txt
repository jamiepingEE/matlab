clear
clc

n=-10:10;
x=zeros(1,21)
x(9:13)=1;

%question1 -pi to pi
w=linspace(-pi,pi,1000)



for a=1:length(w)%every value of w from 1 to some length
    
    sum = 0 ;
    for b = 1:length(n)%for every element of n
        sum = sum + x(b) * exp( -i*w(a)*b )%neg j omega a from x[a] b from X[b]
    end
    sum1(a)=sum;
end

subplot(2,1,1)
stem(n,x)
title('input')

subplot(2,1,2)
plot(w,abs(sum1))%absolute value of sum1
title('DTFT using loop')
